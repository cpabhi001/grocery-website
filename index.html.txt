let lang="en";
let cart=[];

function toggleLang(){
 lang = (lang==="en")?"ml":"en";

function addItem(en,ml,unit,qty){
 qty=parseInt(qty);
 let name = (lang==="ml")?ml:en;

 else cart.push({name,unit,qty});

}

  t.innerHTML+=`<tr>

  <td><button class="remove" onclick="removeItem(${idx})">X</button></td>
  </tr>`;
 });
}
